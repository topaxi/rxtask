    import { Subject } from 'rxjs/Subject';
import { Subscription } from 'rxjs/Subscription';
import { observable } from 'rxjs/symbol/observable';
import { map, filter, materialize, multicast, distinctUntilChanged, } from 'rxjs/operators';
import { ReplaySubject } from 'rxjs/ReplaySubject';
import * as taskInstance from './reducers/task-instance';
import { ObjectUnsubscribedError } from 'rxjs/util/ObjectUnsubscribedError';
import { actionReducer } from './operators';
import { createNotificationAction, } from './actions/task-instance';
/**
 * @class TaskInstance<T>
 * @implements {Subscribable<T>}
 * @implements {ISubscription}
 */
export class TaskInstance {
    /**
     * @param {Observable<T>} observable$
     * @return {TaskInstance<T>}
     */
    constructor(observable$) {
        this._closed = false;
        this._observableMirror$ = new Subject();
        this._currentState$ = new ReplaySubject(1);
        this._subscription = new Subscription();
        this._observableNotifications$ = this._observableMirror$.pipe(materialize());
        this._actions$ = this._observableNotifications$.pipe(map(createNotificationAction));
        /** @type {Observable<TaskInstanceState<T>>} */
        this.state$ = this._actions$.pipe(actionReducer(taskInstance.reducer), multicast(this._currentState$)).refCount();
        /** @type {Observable<TaskInstanceStateLabel<T>>} */
        this.stateLabel$ = this.state$.pipe(map(taskInstance.selectStateLabel), distinctUntilChanged());
        /** @type {Observable<boolean>} */
        this.isPending$ = this.stateLabel$.pipe(map(taskInstance.isPending));
        /** @type {Observable<boolean>} */
        this.isRunning$ = this.stateLabel$.pipe(map(taskInstance.isRunning));
        /** @type {Observable<boolean>} */
        this.isCancelled$ = this.stateLabel$.pipe(map(taskInstance.isCancelled));
        /** @type {Observable<boolean>} */
        this.isError$ = this.stateLabel$.pipe(map(taskInstance.isError));
        /** @type {Observable<boolean>} */
        this.isComplete$ = this.stateLabel$.pipe(map(taskInstance.isComplete));
        /** @type {Observable<boolean>} */
        this.hasValue$ = this.state$.pipe(map(taskInstance.hasValue));
        /** @type {Observable<T>} */
        this.currentValue$ = this.state$.pipe(filter(taskInstance.hasValue), map(taskInstance.selectValue));
        /** @type {Observable<Error|null>} */
        this.error$ = this.state$.pipe(map(taskInstance.selectError));
        this._observable$ = observable$.pipe(multicast(this._observableMirror$)).refCount();
    }
    /** @type {boolean} */
    get closed() {
        return this._closed;
    }
    /** @ignore */
    [observable]() {
        return this;
    }
    /**
     * @param {Observer|Function} observerOrNext (optional) Either an observer with methods to be called,
     *  or the first of three possible handlers, which is the handler for each value emitted from the subscribed
     *  Observable.
     * @param {Function} error (optional) A handler for a terminal event resulting from an error. If no error handler is provided,
     *  the error will be thrown as unhandled.
     * @param {Function} complete (optional) A handler for a terminal event resulting from successful completion.
     * @return {ISubscription} a subscription reference to the registered handlers
     */
    subscribe(observerOrNext, error, complete) {
        if (this._closed)
            throw new ObjectUnsubscribedError();
        this._currentState$.next(taskInstance.RUNNING_STATE);
        this._subscription.add(this._observable$.subscribe(observerOrNext, error, complete));
        return this;
    }
    /**
     * @return {void}
     */
    unsubscribe() {
        this._currentState$.next(taskInstance.CANCELLED_STATE);
        this._closed = true;
        this._subscription.unsubscribe();
    }
}
