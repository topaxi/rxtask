export const INITIAL_STATE = {
    performed: 0,
    pending: EMPTY_ARRAY,
    running: EMPTY_ARRAY,
    successful: 0,
    cancelled: 0,
    errored: 0,
    completed: 0,
    last: null,
    lastRunning: null,
    lastSuccessful: null,
    lastCancelled: null,
    lastErrored: null,
    lastCompleted: null,
};
function taskInstanceStateReducer(state, { taskInstance, taskInstanceStateLabel }) {
    switch (taskInstanceStateLabel) {
        case "pending" /* PENDING */:
            return Object.assign({}, state, { performed: state.performed + 1, pending: [...state.pending, taskInstance], last: taskInstance });
        case "running" /* RUNNING */:
            return Object.assign({}, taskNoLonger(state, taskInstance, 'pending'), { running: [...state.running, taskInstance], lastRunning: taskInstance });
        case "cancelled" /* CANCELLED */:
            return Object.assign({}, pipe(state, state => taskNoLonger(state, taskInstance, 'pending'), state => taskNoLonger(state, taskInstance, 'running')), { cancelled: state.cancelled + 1, lastCancelled: taskInstance });
        case "error" /* ERROR */:
            return Object.assign({}, taskCompleted(state, taskInstance), { errored: state.errored + 1, lastErrored: taskInstance });
        case "complete" /* COMPLETE */:
            return Object.assign({}, taskCompleted(state, taskInstance), { successful: state.successful + 1, lastSuccessful: taskInstance });
        default:
            return assertNever(taskInstanceStateLabel);
    }
}
/**
 * @param {TaskState<T>} [state=INITIAL_STATE]
 * @param {TaskActions<T>} action
 * @return {TaskState<T>}
 */
export function reducer(state = INITIAL_STATE, action) {
    switch (action.type) {
        case taskActions.TASK_INSTANCE_STATE_UPDATE_ACTION:
            return taskInstanceStateReducer(state, action.payload);
        default:
            return state;
    }
}
function taskNoLonger(state, task, taskState) {
    return Object.assign({}, state, { [taskState]: state[taskState].filter(neq(task)) });
}
function taskCompleted(state, task) {
    return Object.assign({}, pipe(state, state => taskNoLonger(state, task, 'pending'), state => taskNoLonger(state, task, 'running')), { completed: state.completed + 1, lastCompleted: task });
}
