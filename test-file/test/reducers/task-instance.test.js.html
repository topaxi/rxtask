    test/reducers/task-instance.test.js

    import { Notification } from 'rxjs/Notification';
import * as taskInstance from '../../src/reducers/task-instance';
import { createNotificationAction, NOTIFICATION_ACTION, } from '../../src/actions/task-instance';
import { EMPTY_ACTION } from './helpers';
import { expect } from 'chai';
function createNextNotificationAction(value) {
    return createNotificationAction(Notification.createNext(value));
}
function createErrorNotificationAction(value) {
    return createNotificationAction(Notification.createError(value));
}
function createCompleteNotificationAction() {
    return createNotificationAction(Notification.createComplete());
}
describe('reducers/TaskInstance', () => {
    describe('Static states', () => {
        /** @test {PENDING_STATE} */
        it('defines a pending state', () => {
            expect(taskInstance.PENDING_STATE).to.deep.equal({
                stateLabel: "pending" /* PENDING */,
                hasValue: false,
                currentValue: undefined,
                error: null,
            });
        });
        /** @test {RUNNING_STATE} */
        it('defines a running state', () => {
            expect(taskInstance.RUNNING_STATE).to.deep.equal({
                stateLabel: "running" /* RUNNING */,
                hasValue: false,
                currentValue: undefined,
                error: null,
            });
        });
        /** @test {CANCELLED_STATE} */
        it('defines a cancelled state', () => {
            expect(taskInstance.CANCELLED_STATE).to.deep.equal({
                stateLabel: "cancelled" /* CANCELLED */,
                hasValue: false,
                currentValue: undefined,
                error: null,
            });
        });
    });
    /** @test {reducer} */
    describe('Reducer', () => {
        describe('Initial state', () => {
            it('equals PENDING_STATE', () => {
                expect(taskInstance.reducer(undefined, EMPTY_ACTION)).to.deep.equal(taskInstance.PENDING_STATE);
            });
        });
        describe('Notifications', () => {
            let state;
            beforeEach(() => {
                state = taskInstance.PENDING_STATE;
            });
            describe('Next Notification', () => {
                beforeEach(() => {
                    state = taskInstance.reducer(state, createNextNotificationAction('VALUE'));
                });
                it('sets state to running', () => {
                    expect(state.stateLabel).to.equal("running" /* RUNNING */);
                });
                it('marks state to having a value', () => {
                    expect(state.hasValue).to.be.true;
                });
                it('sets current value', () => {
                    expect(state.currentValue).to.equal('VALUE');
                    state = taskInstance.reducer(state, createNextNotificationAction('NEXT VALUE'));
                    expect(state.currentValue).to.equal('NEXT VALUE');
                });
                it('does not modify error state', () => {
                    state = taskInstance.reducer({ error: 'error' }, createNextNotificationAction('NEXT VALUE'));
                    expect(state.error).to.equal('error');
                });
            });
            describe('Error Notification', () => {
                beforeEach(() => {
                    state = taskInstance.reducer(state, createErrorNotificationAction('ERROR'));
                });
                it('sets state to error', () => {
                    expect(state.stateLabel).to.equal("error" /* ERROR */);
                });
                it('does not modify hasValue', () => {
                    expect(state.hasValue).to.be.false;
                    state = taskInstance.reducer({ hasValue: true }, createErrorNotificationAction('ERROR'));
                    expect(state.hasValue).to.be.true;
                });
                it('does not modify current value', () => {
                    expect(state.currentValue).to.be.undefined;
                    state = taskInstance.reducer({ currentValue: true }, createErrorNotificationAction('ERROR'));
                    expect(state.currentValue).to.be.true;
                });
                it('sets error property', () => {
                    expect(state.error).to.equal('ERROR');
                });
            });
            describe('Complete Notification', () => {
                beforeEach(() => {
                    state = taskInstance.reducer(state, createCompleteNotificationAction());
                });
                it('sets state to complete', () => {
                    expect(state.stateLabel).to.equal("complete" /* COMPLETE */);
                });
                it('does not modify hasValue', () => {
                    expect(state.hasValue).to.be.false;
                    state = taskInstance.reducer({ hasValue: true }, createCompleteNotificationAction());
                    expect(state.hasValue).to.be.true;
                });
                it('does not modify current value', () => {
                    expect(state.currentValue).to.be.undefined;
                    state = taskInstance.reducer({ currentValue: true }, createCompleteNotificationAction());
                    expect(state.currentValue).to.be.true;
                });
                it('does not modify error state', () => {
                    state = taskInstance.reducer({ error: 'error' }, createCompleteNotificationAction());
                    expect(state.error).to.equal('error');
                });
            });
            describe('Unknown Notification', () => {
                beforeEach(() => {
                    state = taskInstance.reducer(state, {
                        type: NOTIFICATION_ACTION,
                        payload: { kind: '__UNKOWN' },
                    });
                });
                it('does not modify any state', () => {
                    expect(state).to.deep.equal(taskInstance.PENDING_STATE);
                });
            });
        });
    });
    describe('Filters', () => {
        /** @test {isPending} */
        it('isPending', () => {
            expect(taskInstance.isPending("pending" /* PENDING */)).to.be.true;
            expect(taskInstance.isPending("running" /* RUNNING */)).to.be
                .false;
            expect(taskInstance.isPending("complete" /* COMPLETE */)).to.be
                .false;
            expect(taskInstance.isPending("cancelled" /* CANCELLED */)).to.be
                .false;
            expect(taskInstance.isPending("error" /* ERROR */)).to.be.false;
        });
        /** @test {isRunning} */
        it('isRunning', () => {
            expect(taskInstance.isRunning("pending" /* PENDING */)).to.be
                .false;
            expect(taskInstance.isRunning("running" /* RUNNING */)).to.be.true;
            expect(taskInstance.isRunning("complete" /* COMPLETE */)).to.be
                .false;
            expect(taskInstance.isRunning("cancelled" /* CANCELLED */)).to.be
                .false;
            expect(taskInstance.isRunning("error" /* ERROR */)).to.be.false;
        });
        /** @test {isComplete} */
        it('isComplete', () => {
            expect(taskInstance.isComplete("pending" /* PENDING */)).to.be
                .false;
            expect(taskInstance.isComplete("running" /* RUNNING */)).to.be
                .false;
            expect(taskInstance.isComplete("complete" /* COMPLETE */)).to.be
                .true;
            expect(taskInstance.isComplete("cancelled" /* CANCELLED */)).to.be
                .false;
            expect(taskInstance.isComplete("error" /* ERROR */)).to.be.false;
        });
        /** @test {isCancelled} */
        it('isCancelled', () => {
            expect(taskInstance.isCancelled("pending" /* PENDING */)).to.be
                .false;
            expect(taskInstance.isCancelled("running" /* RUNNING */)).to.be
                .false;
            expect(taskInstance.isCancelled("complete" /* COMPLETE */)).to.be
                .false;
            expect(taskInstance.isCancelled("cancelled" /* CANCELLED */)).to.be
                .true;
            expect(taskInstance.isCancelled("error" /* ERROR */)).to.be
                .false;
        });
        /** @test {isError} */
        it('isError', () => {
            expect(taskInstance.isError("pending" /* PENDING */)).to.be.false;
            expect(taskInstance.isError("running" /* RUNNING */)).to.be.false;
            expect(taskInstance.isError("complete" /* COMPLETE */)).to.be.false;
            expect(taskInstance.isError("cancelled" /* CANCELLED */)).to.be
                .false;
            expect(taskInstance.isError("error" /* ERROR */)).to.be.true;
        });
    });
    describe('Selectors', () => {
        /** @test {selectValue} */
        it('selectValue', () => {
            expect(taskInstance.selectValue({ currentValue: 'VALUE' })).to.equal('VALUE');
        });
        /** @test {selectHasValue} */
        it('selectHasValue', () => {
            expect(taskInstance.selectHasValue({ hasValue: true })).to.be.true;
        });
        /** @test {selectState} */
        it('selectState', () => {
            expect(taskInstance.selectStateLabel({
                stateLabel: "error" /* ERROR */,
            })).to.equal("error" /* ERROR */);
        });
        /** @test {selectError} */
        it('selectError', () => {
            expect(taskInstance.selectError({ error: 'ERROR' })).to.equal('ERROR');
        });
    });
    describe('helpers', () => {
        /** @test {hasValue} */
        it('hasValue', () => {
            const hasValue = [
                { hasValue: true, stateLabel: "running" /* RUNNING */ },
                { hasValue: true, stateLabel: "complete" /* COMPLETE */ },
            ];
            const hasNoValue = [
                { hasValue: false, stateLabel: "pending" /* PENDING */ },
                { hasValue: false, stateLabel: "running" /* RUNNING */ },
                { hasValue: false, stateLabel: "cancelled" /* CANCELLED */ },
                { hasValue: false, stateLabel: "complete" /* COMPLETE */ },
                { hasValue: false, stateLabel: "error" /* ERROR */ },
                // As something went wrong, we assume we don't have a valid value
                { hasValue: true, stateLabel: "error" /* ERROR */ },
                // Cancelled task values probably don't make sense
                { hasValue: true, stateLabel: "cancelled" /* CANCELLED */ },
                // Tasks have no initial value
                { hasValue: true, stateLabel: "pending" /* PENDING */ },
            ];
            const expectHasValue = (has) => (v) => expect(taskInstance.hasValue(v), JSON.stringify(v)).to.equal(has);
            hasValue.forEach(expectHasValue(true));
            hasNoValue.forEach(expectHasValue(false));
        });
    });
});
